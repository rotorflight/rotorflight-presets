#$ TITLE: Hobbywing V4 telemetry setup
#$ FIRMWARE_VERSION: 4.5
#$ FIRMWARE_VERSION: 4.6
#$ BOARD_NAME: FLYDRAGON_PRO6000
#$ BOARD_NAME: FLYDRAGON_PRO42688


#$ CATEGORY: SETUP
#$ STATUS: EXPERIMENTAL
#$ KEYWORDS: ESC Telemetry
#$ AUTHOR: Hans Ouwehand
#$ DESCRIPTION: Enables ESC telemetry from the Hobbywing V4 esc. Make sure you have the latest hobbywing firmware but not the vbar hobbywing firmware.
#$ DESCRIPTION: Hobbywing v4 esc's does not support forward programming.
#$ DESCRIPTION: Choose the port where you have connected the esc telemetry wire
#$ DESCRIPTION: Make sure to use only signal from the p port and eventually ground. Don't connect the plus wire.
#$ PRIORITY: 0
#$ FORCE_OPTIONS_REVIEW: TRUE


feature ESC_SENSOR
feature FREQ_SENSOR
set current_meter = ESC
set battery_meter = ESC
set esc_sensor_protocol = HOBBYWINGV4
set esc_sensor_halfduplex = off



#$ OPTION_GROUP BEGIN: (EXCLUSIVE) Port
    #$ OPTION BEGIN (UNCHECKED): dsm
    serial 2 1024 115200 57600 0 115200
    set esc_sensor_pinswap = off
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): TX2
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = on
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): RX2
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = off
    #$ OPTION END
#$ OPTION_GROUP END
