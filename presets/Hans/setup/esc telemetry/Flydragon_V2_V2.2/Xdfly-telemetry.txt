#$ TITLE: Xdfly telemetry setup
#$ FIRMWARE_VERSION: 4.5
#$ FIRMWARE_VERSION: 4.6
#$ BOARD_NAME: FLYDRAGON_V2
#$ BOARD_NAME: FLYDRAGON_V2_2


#$ CATEGORY: SETUP
#$ STATUS: EXPERIMENTAL
#$ KEYWORDS: ESC Telemetry, XDfly
#$ AUTHOR: Hans Ouwehand
#$ DESCRIPTION: Enables ESC telemetry from the XDfly esc. 
#$ DESCRIPTION: Make sure to use only signal from the esc Rx pin and eventually ground. Choose the port where you have connected the telemetry wire.
#$ PRIORITY: 0


feature ESC_SENSOR
feature FREQ_SENSOR
set current_meter = ESC
set battery_meter = ESC
set esc_sensor_protocol = XDFLY
set esc_sensor_halfduplex = ON



#$ OPTION_GROUP BEGIN: (EXCLUSIVE) Port
    #$ OPTION BEGIN (UNCHECKED): F.port
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = Off
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): sbus
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = ON
    #$ OPTION END
#$ OPTION_GROUP END
