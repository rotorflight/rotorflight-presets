#$ DESCRIPTION: Enables ESC telemetry from the scorpion esc. Make sure you have selected unsollicited as telemetry protocol in the Scorpion esc.

#$ DESCRIPTION: Make sure to use only signal from the pc port and eventually ground. Don't connect the plus wire.

#$ FORCE_OPTIONS_REVIEW: TRUE


feature ESC_SENSOR
feature FREQ_SENSOR
set current_meter = ESC
set battery_meter = ESC
set esc_sensor_protocol = SCORPION
set esc_sensor_halfduplex = ON



#$ OPTION_GROUP BEGIN: (EXCLUSIVE) Port
    #$ OPTION BEGIN (UNCHECKED): F.port
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = Off
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): sbus
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = ON
    #$ OPTION END
