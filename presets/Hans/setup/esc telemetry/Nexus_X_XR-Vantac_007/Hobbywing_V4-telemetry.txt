#$ TITLE: Hobbywing V4 telemetry setup
#$ FIRMWARE_VERSION: 4.5
#$ FIRMWARE_VERSION: 4.6
#$ BOARD_NAME: VANTAC_RF007
#$ BOARD_NAME: NEXUS_X
#$ BOARD_NAME: NEXUS_XR


#$ CATEGORY: SETUP
#$ STATUS: EXPERIMENTAL
#$ KEYWORDS: ESC Telemetry
#$ AUTHOR: Hans Ouwehand
#$ DESCRIPTION: Enables ESC telemetry from the Hobbywing V4 esc. Make sure you have the latest hobbywing firmware but not the vbar hobbywing firmware.
#$ DESCRIPTION: Hobbywing v4 esc's does not support forward programming.
#$ DESCRIPTION: Make sure to use only signal from the p port and eventually ground. Connect it to the tlm port.
#$ PRIORITY: 0


feature ESC_SENSOR
feature FREQ_SENSOR
serial 1 1024 115200 57600 0 115200
set current_meter = ESC
set battery_meter = ESC
set esc_sensor_protocol = HOBBYWINGV4
set esc_sensor_halfduplex = off
set esc_sensor_pinswap = off
