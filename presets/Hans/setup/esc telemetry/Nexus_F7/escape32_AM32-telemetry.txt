#$ TITLE: Escape32/AM32 telemetry setup
#$ FIRMWARE_VERSION: 4.5
#$ FIRMWARE_VERSION: 4.6
#$ BOARD_NAME: FLYDRAGON_PRO6000
#$ BOARD_NAME: FLYDRAGON_PRO42688


#$ CATEGORY: SETUP
#$ STATUS: COMMUNITY
#$ KEYWORDS: ESC Telemetry
#$ AUTHOR: Hans Ouwehand
#$ DESCRIPTION: Enables ESC telemetry from a Escape32/AM32 esc. Make sure you have selected kiss auto as telemetry protocol in the Escape32/AM32 esc.
#$ DESCRIPTION: Choose the port on the fc where you have connected the esc telemetry wire
#$ DESCRIPTION: Make sure to use only signal wire from a telemetry pad.
#$ PRIORITY: 0
#$ FORCE_OPTIONS_REVIEW: TRUE


feature ESC_SENSOR
set current_meter = ESC
set battery_meter = ESC
set esc_sensor_protocol = BLHELI32
set esc_sensor_halfduplex = off



#$ OPTION_GROUP BEGIN: (EXCLUSIVE) Port
    #$ OPTION BEGIN (UNCHECKED): dsm
    serial 0 1024 115200 57600 0 115200
    set esc_sensor_pinswap = off
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): SBUS
    serial 1 1024 115200 57600 0 115200
    set esc_sensor_pinswap = off
    #$ OPTION END
#$ OPTION_GROUP END
