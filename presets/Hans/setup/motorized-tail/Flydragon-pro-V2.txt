#$ DESCRIPTION: Set the parameters for the motorized tail. It remaps the tail channel to motor2 so the tail can be controlled as a motorized tail
#$ DESCRIPTION: Also in the mixer page it sets the tail rotor mode to motorized

#$ PRIORITY: 0

resource servo 4 none
resource motor2 C09
set tail_rotor_mode = MOTORIZED
