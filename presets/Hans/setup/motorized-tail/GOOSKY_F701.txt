#$ TITLE: Set parameters motorized tail
#$ FIRMWARE_VERSION: 4.5
#$ FIRMWARE_VERSION: 4.6
#$ BOARD_NAME: GOOSKY_F701

#$ CATEGORY: SETUP
#$ STATUS: OFFICIAL
#$ KEYWORDS: Tail, Setup, Motorized tail
#$ AUTHOR: Paul Lesogor
#$ DESCRIPTION: Set the parameters for the motorized tail. It remaps the tail channel to motor 2 so the tail can be controlled as a motorized tail
#$ DESCRIPTION: Also in the mixer page it sets the tail rotor mode to motorized

#$ PRIORITY: 0

resource servo 4 none
resource motor 2 A09
set tail_rotor_mode = MOTORIZED
