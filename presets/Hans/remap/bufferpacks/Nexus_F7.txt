#$ TITLE: Remap a port to switch on/off capacitor pack
#$ FIRMWARE_VERSION: 4.4

#$ CATEGORY: REMAPPING
#$ STATUS: OFFICIAL

#$ DESCRIPTION: Enables a port to enable/disable the bufferpack.
#$ DESCRIPTION: Remap the sbus port to switch on/off the bufferpack.
#$ DESCRIPTION: You can choose the radio channel which you want to use to switch it on/off
#$ PRIORITY: 0
#$ FORCE_OPTIONS_REVIEW: TRUE

resource SERIAL_RX 2 NONE
resource SERVO 5 A03

#$ OPTION_GROUP BEGIN: (EXCLUSIVE) Input Channel
    #$ OPTION BEGIN (UNCHECKED): Channel 7
    mixer input AUX2 -1000 1000 1000
    mixer rule 10 set AUX2 S5 1000 0
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): Channel 8
    mixer input AUX3 -1000 1000 1000
    mixer rule 10 set AUX3 S5 1000 0
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): Channel 9
    mixer input CH9 -1000 1000 1000
    mixer rule 10 set CH9 S5 1000 0
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): Channel 10
    mixer input CH10 -1000 1000 1000

    #$ OPTION END

    #$ OPTION END
